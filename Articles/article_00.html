<script>
let editorInstance;

// Initialize CKEditor
ClassicEditor
   .create(document.querySelector('#editor'), {
       toolbar: [
           'heading', '|',
           'bold', 'italic', 'underline', 'strikethrough', '|',
           'fontColor', 'fontBackgroundColor', 'fontSize', 'fontFamily', '|',
           'alignment', '|',
           'bulletedList', 'numberedList', '|',
           'link', 'insertImage', 'insertTable', '|',
           'undo', 'redo'
       ]
   })
   .then(editor => {
       editorInstance = editor;
       console.log('Editor initialized');
   })
   .catch(error => {
       console.error('CKEditor initialization failed:', error);
   });

// Insert image from URL
document.getElementById('insertImageBtn').addEventListener('click', function() {
   if (!editorInstance) {
       alert('Editor not ready yet. Please wait a moment.');
       return;
   }

   const imageUrl = prompt("Please enter the image URL:");

   if (imageUrl && imageUrl.trim() !== "") {
       const url = imageUrl.trim();

       // Optional: basic check before inserting
       const testImg = new Image();
       testImg.onload = () => {
           // Image loads successfully → insert it
           editorInstance.execute('insertImage', {
               source: url
           });
       };
       testImg.onerror = () => {
           alert("Cannot load this image.\n\n" +
                 "Possible reasons:\n" +
                 "• The link is not a direct image (must end with .jpg, .png, .gif, etc.)\n" +
                 "• The image is blocked (anti-hotlinking)\n" +
                 "• The URL is invalid or the image was deleted\n\n" +
                 "Try another link or use a public image host like imgur.com");
       };
       testImg.src = url;  // start loading test
   }
});

// Save article as HTML file
document.getElementById('saveBtn').addEventListener('click', function() {
   if (!editorInstance) {
       alert('Editor not ready yet.');
       return;
   }

   const content = editorInstance.getData();

   const blob = new Blob([content], { type: "text/html" });
   const url = URL.createObjectURL(blob);

   const a = document.createElement('a');
   a.href = url;
   a.download = 'article.html';
   a.click();

   URL.revokeObjectURL(url);

   alert('Article saved as article.html');
});
</script>
